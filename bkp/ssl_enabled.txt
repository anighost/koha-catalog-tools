  Step 1: Expand the Certificate to Cover the Staff Subdomain

  sudo certbot certonly --apache \
    -d admin.library.disharifoundation.org

  ---
  Step 2: Edit dishari_lib-le-ssl.conf

  Add the Staff HTTPS VHost inside the existing <IfModule mod_ssl.c> block, right after the closing </VirtualHost> of OPAC:

  <IfModule mod_ssl.c>
  <VirtualHost *:443>
    <IfVersion >= 2.4>
     Define instance "dishari_lib"
    </IfVersion>
     Include /etc/koha/apache-shared.conf
  #  Include /etc/koha/apache-shared-disable.conf
    Include /etc/koha/apache-shared-opac-plack.conf
     Include /etc/koha/apache-shared-opac.conf

     ServerName library.disharifoundation.org
     ServerAlias www.library.disharifoundation.org koha.disharifoundation.org
     SetEnv KOHA_CONF "/etc/koha/sites/dishari_lib/koha-conf.xml"
     AssignUserID dishari_lib-koha dishari_lib-koha

     ErrorLog /var/log/koha/dishari_lib/opac-error.log

  SSLCertificateFile /etc/letsencrypt/live/library.disharifoundation.org/fullchain.pem
  SSLCertificateKeyFile /etc/letsencrypt/live/library.disharifoundation.org/privkey.pem
  Include /etc/letsencrypt/options-ssl-apache.conf
  </VirtualHost>

  # Staff HTTPS  ← ADD THIS BLOCK
  <VirtualHost *:443>
    <IfVersion >= 2.4>
      Define instance "dishari_lib"
    </IfVersion>
    Include /etc/koha/apache-shared.conf
    Include /etc/koha/apache-shared-intranet-plack.conf
    Include /etc/koha/apache-shared-intranet.conf

    ServerName admin.library.disharifoundation.org
    SetEnv KOHA_CONF "/etc/koha/sites/dishari_lib/koha-conf.xml"
    AssignUserID dishari_lib-koha dishari_lib-koha

    ErrorLog /var/log/koha/dishari_lib/intranet-error.log

  SSLCertificateFile /etc/letsencrypt/live/library.disharifoundation.org/fullchain.pem
  SSLCertificateKeyFile /etc/letsencrypt/live/library.disharifoundation.org/privkey.pem
  Include /etc/letsencrypt/options-ssl-apache.conf
  </VirtualHost>
  </IfModule>

  Step 3: Edit dishari_lib.conf — Add Staff HTTP Redirect, Clean Up :8080

  Add a redirect for the staff subdomain and replace the :8080 block (which is no longer needed):

  # OPAC HTTP → HTTPS
  <VirtualHost *:80>
    <IfVersion >= 2.4>
     Define instance "dishari_lib"
    </IfVersion>
     Include /etc/koha/apache-shared.conf
     Include /etc/koha/apache-shared-opac-plack.conf
     Include /etc/koha/apache-shared-opac.conf

     ServerName library.disharifoundation.org
     ServerAlias www.library.disharifoundation.org koha.disharifoundation.org
     SetEnv KOHA_CONF "/etc/koha/sites/dishari_lib/koha-conf.xml"
     AssignUserID dishari_lib-koha dishari_lib-koha

     ErrorLog    /var/log/koha/dishari_lib/opac-error.log
  RewriteEngine on
  RewriteCond %{SERVER_NAME} =koha.disharifoundation.org [OR]
  RewriteCond %{SERVER_NAME} =library.disharifoundation.org [OR]
  RewriteCond %{SERVER_NAME} =www.library.disharifoundation.org
  RewriteRule ^ https://%{SERVER_NAME}%{REQUEST_URI} [END,NE,R=permanent]
  </VirtualHost>

  # Staff HTTP → HTTPS  ← ADD THIS BLOCK
  <VirtualHost *:80>
    ServerName admin.library.disharifoundation.org
    RewriteEngine on
    RewriteRule ^ https://%{SERVER_NAME}%{REQUEST_URI} [END,NE,R=permanent]
  </VirtualHost>

  ▎ The old <VirtualHost *:8080> block is no longer needed — staff is now served on :443.
